# mean ans sd functions
funm <- function(x){mean(x, na.rm = T)}

#table
dogsWH <- data_raw@phdata %>%
  group_by(Breed, sex) %>%
  summarize(n = n(),
            mean.Age = funm(Age), 
            mean.Weight = funm(Body_weight)) 

dogsWH$sex[dogsWH$sex == 0]<- "female"
dogsWH$sex[dogsWH$sex == 1]<- "male"

print(dogsWH)
